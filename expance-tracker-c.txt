#include <stdio.h>
#include <stdlib.h>

struct Expense {
    float amount;
    char category[50];
    char note[100];
};

void addExpense() {
    FILE *file = fopen("expenses.txt", "a");
    struct Expense e;

    printf("Enter amount: ");
    scanf("%f", &e.amount);

    printf("Enter category: ");
    scanf("%s", e.category);

    printf("Enter note: ");
    scanf("%s", e.note);

    fprintf(file, "%.2f %s %s\n", e.amount, e.category, e.note);

    fclose(file);

    printf("Expense added successfully!\n");
}

void viewExpenses() {
    FILE *file = fopen("expenses.txt", "r");
    struct Expense e;

    printf("\n--- Expense List ---\n");

    while (fscanf(file, "%f %s %s", &e.amount, e.category, e.note) != EOF) {
        printf("Amount: %.2f | Category: %s | Note: %s\n", e.amount, e.category, e.note);
    }

    fclose(file);
}

int main() {
    int choice;

    while (1) {
        printf("\n==== Personal Expense Tracker ====\n");
        printf("1. Add Expense\n");
        printf("2. View Expenses\n");
        printf("3. Exit\n");
        printf("Enter choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                addExpense();
                break;
            case 2:
                viewExpenses();
                break;
            case 3:
                exit(0);
            default:
                printf("Invalid choice\n");
        }
    }

    return 0;
}
